html code

<div class="text-center"> <div class="pagination"></div> </div>

1. simple example

$('.pagination1').bsPagination({ totalRecord: 200, recentPage: 1 });

2. add more config params

$('.pagination2').bsPagination({ totalRecord: 400, recentPage: 15, recordPerPage: 10, pagePerBatch: 10, showFirstPage: true, showLastPage: true, firstPageWord: '1st page', lastPageWord: 'last page' });

3. set page link to change page, the url will change when you click page

$('.pagination3').bsPagination({ totalRecord: 200, recentPage: 8, setLinkAttr: function (page) { return 'href="#/page/' + page + '"'; } });

4. set click link event to change page

$('.pagination4').bsPagination({ totalRecord: 200, recentPage: 18 }); $('.pagination4').on('click', 'a', function () { $('.pagination4').bsPagination({ totalRecord: 200, recentPage: $(this).data('page') }); });
